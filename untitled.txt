# عرض محتويات مجلد workspace
ls -la /data/chats/11c2b2cfe560470bbac538585b1d81a2/workspace/

# الدخول إلى مجلد react_template وعرض محتوياته
cd /data/chats/11c2b2cfe560470bbac538585b1d81a2/workspace/react_template && ls -la
